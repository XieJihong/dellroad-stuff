008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.SessionDestroyEvent;
011import com.vaadin.server.SessionDestroyListener;
012import com.vaadin.server.SessionInitEvent;
013import com.vaadin.server.SessionInitListener;
014import com.vaadin.server.VaadinService;
015import com.vaadin.server.VaadinSession;
016
017import java.util.ArrayList;
018import java.util.Collection;
019
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023/**
024 * Container containing active {@link VaadinSession}s. This class is useful when you need to display
025 * sessions in your GUI, for example, a table showing all logged-in users.
026 *
027 * <p>
028 * By subclassing this class and {@link VaadinSessionInfo}, additional session-related properties,
029 * such as logged-in user, can be added to the container by annotating the {@link VaadinSessionInfo}
030 * subclass with {@link ProvidesProperty &#64;ProvidesProperty} annotations.
031 * </p>
032 *
033 * <p>
034 * Note the following:
035 * <ul>
036 *  <li>The {@link SpringVaadinServlet} must be used and configured with init parameter
037 *      {@link SpringVaadinServlet#SESSION_TRACKING_PARAMETER} set to {@code true}.</li>
038 *  <li>The {@link #connect} and {@link #disconnect} methods must be invoked before and after (respectively)
039 *      this container is used; typically these would be invoked in the {@link com.vaadin.ui.Component#attach}
040 *      and {@link com.vaadin.ui.Component#detach} methods of a corresponding widget.
041 * </ul>
042 * </p>
043 *
044 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
045 */
046@SuppressWarnings("serial")
047public abstract class VaadinSessionContainer<T extends VaadinSessionInfo> extends SimpleKeyedContainer<VaadinSession, T> {
048
049    /**
050     * The {@link VaadinSession} that this container instance is associated with.
051     * Determined by whatever {@link VaadinSession} is associated with the current thread at construction time.
052     */
053    protected final VaadinSession session;
054
055    private final Logger log = LoggerFactory.getLogger(this.getClass());
056    private final SessionEventListener listener = new SessionEventListener();
057
058    /**
059     * Constructor.
060     *
061     * <p>
062     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
063     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
064     * </p>
065     *
066     * @param type class to introspect for annotated methods
067     * @throws IllegalArgumentException if {@code type} is null
068     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
069     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
070     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
071     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
072     *  property "getter" method
073     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
074     * @throws IllegalStateException if there is no {@link SpringVaadinServlet} associated
075     *  with the current thread's {@link VaadinSession}
076     * @see ProvidesProperty
077     * @see ProvidesPropertySort
078     * @see ProvidesPropertyScanner
079     */
080    protected VaadinSessionContainer(Class<T> type) {
081        super(type);
082        this.session = VaadinUtil.getCurrentSession();
083        SpringVaadinServlet.getServlet(this.session);
084    }
085
086    /**
087     * Constructor.
088     *
089     * @param propertyExtractor used to extract properties from the underlying Java objects;
090     *  may be null but then container is not usable until one is configured via
091     * {@link #setPropertyExtractor setPropertyExtractor()}
092     * @param propertyDefs container property definitions; null is treated like the empty set
093     */
094    protected VaadinSessionContainer(PropertyExtractor<? super T> propertyExtractor,
095      Collection<? extends PropertyDef<?>> propertyDefs) {
096        super(propertyExtractor, propertyDefs);
097        this.session = VaadinUtil.getCurrentSession();
098        SpringVaadinServlet.getServlet(this.session);
099    }
100
101    @Override
102    protected VaadinSession getKeyFor(T info) {
103        return info.getVaadinSession();
104    }
105
106    /**
107     * Start tracking live sessions.
108     *
109     */
110    public void connect() {
111        this.listener.register();
112        this.reload();
113    }
114
115    /**
116     * Stop tracking live sessions.
117     */
118    public void disconnect() {
119        this.listener.unregister();
120    }
121
122    /**
123     * Reload this container.
124     *
125     * <p>
126     * This method can be invoked from any thread. It creates a new thread to do the actual
127     * reloading to avoid potential deadlocks.
128     * </p>
129     */
130    public void reload() {
131        new Thread("VaadinSessionContainer.reload()") {
132            @Override
133            public void run() {
134                try {
135                    VaadinSessionContainer.this.update();
136                } catch (ThreadDeath t) {
137                    throw t;
138                } catch (Throwable t) {
139                    VaadinSessionContainer.this.log.error("error updating container " + this, t);
140                }
141            }
142        }.start();
143    }
144
145    /**
146     * Update this container after a {@link VaadinSession} has been added or removed.
147     * This must <b>not</b> be invoked while any {@link VaadinSession} is locked.
148     */
149    protected void update() {
150
151        // Sanity check
152        if (VaadinSession.getCurrent() != null && VaadinSession.getCurrent().hasLock())
153            throw new IllegalStateException("inside locked VaadinSession");
154
155        // Create a VaadinSessionInfo object for each session, but doing so while that session is locked
156        final ArrayList<T> sessionInfoList = new ArrayList<T>();
157        for (VaadinSession otherSession : SpringVaadinServlet.getServlet(this.session).getSessions()) {
158            otherSession.accessSynchronously(new Runnable() {
159                @Override
160                public void run() {
161                    sessionInfoList.add(VaadinSessionContainer.this.createVaadinSessionInfo());
162                }
163            });
164        }
165
166        // Reload this container
167        this.session.access(new Runnable() {
168            @Override
169            public void run() {
170                VaadinSessionContainer.this.load(sessionInfoList);
171            }
172        });
173    }
174
175    /**
176     * Create a {@link VaadinSessionInfo} backing object for the {@link VaadinSession} associated with the current thread.
177     * The {@link VaadinSession} will be already locked.
178     */
179    protected abstract T createVaadinSessionInfo();
180
181// SessionEventListener
182
183    private class SessionEventListener extends VaadinExternalListener<VaadinService>
184      implements SessionInitListener, SessionDestroyListener {
185
186        public SessionEventListener() {
187            super(VaadinUtil.getCurrentSession().getService());
188        }
189
190        @Override
191        public void sessionInit(SessionInitEvent event) {
192            VaadinSessionContainer.this.reload();
193        }
194
195        @Override
196        public void sessionDestroy(SessionDestroyEvent event) {
197            VaadinSessionContainer.this.reload();
198        }
199
200        @Override
201        protected void register(VaadinService vaadinService) {
202            vaadinService.addSessionInitListener(this);
203            vaadinService.addSessionDestroyListener(this);
204        }
205
206        @Override
207        protected void unregister(VaadinService vaadinService) {
208            vaadinService.removeSessionInitListener(this);
209            vaadinService.removeSessionDestroyListener(this);
210        }
211    }
212}
