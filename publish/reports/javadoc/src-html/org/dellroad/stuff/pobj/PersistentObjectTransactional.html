008package org.dellroad.stuff.pobj;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Inherited;
013import java.lang.annotation.Retention;
014import java.lang.annotation.RetentionPolicy;
015import java.lang.annotation.Target;
016
017/**
018 * Declares that annotated methods should execute within {@link PersistentObjectTransactionManager} transactions.
019 * During the execution, the method has private, atomic access to the associated database.
020 *
021 * <p>
022 * When the associated AOP aspect is applied, annotated methods are executed within a {@link PersistentObjectTransactionManager}
023 * transaction. In normal mode, the transaction simply executes in the current thread. In "Disruptor" mode, the
024 * annotated method executes on the dedicated transaction thread, and the thread that invoked the method will block
025 * until the transaction completes (unless specified as {@link #async}); return values and thrown exceptions are
026 * delivered back to the calling thread as they normally would be.
027 * </p>
028 *
029 * <p>
030 * The associated AOP aspect must be configured with the {@link PersistentObjectTransactionManager}(s) to be
031 * used for these transactional methods. The easiest way to do this is by declaring the aspect in your Spring
032 * bean factory, for example:
033 * <pre>
034 *  &lt;bean class="org.dellroad.stuff.pobj.PersistentObjectTransactionalAspect" factory-method="aspectOf"
035 *      p:persistentObjectTransactionManager-ref="myManagerBean"/&gt;
036 * </pre>
037 * </p>
038 *
039 * <p>
040 * More than one {@link PersistentObjectTransactionManager}s may be configured. For example:
041 * <pre>
042 *  &lt;bean class="org.dellroad.stuff.pobj.PersistentObjectTransactionalAspect" factory-method="aspectOf"&gt;
043 *      &lt;property name="persistentObjectTransactionManagers"&gt;
044 *          &lt;util:list&gt;
045 *              &lt;bean ref="fooManager"/&gt;
046 *              &lt;bean ref="barManager"/&gt;
047 *          &lt;/util:list&gt;
048 *      &lt;/property&gt;
049 *  &lt;/bean&gt;
050 * </pre>
051 * In the case of multiple {@link PersistentObjectTransactionManager}s, which {@link PersistentObjectTransactionManager}
052 * the aspect uses is determined by the {@link #managerName} annotation property, which must equal the
053 * {@linkplain PersistentObjectTransactionManager#getName name} of the desired {@link PersistentObjectTransactionManager}, e.g.:
054 * <pre>
055 *
056 *  &#64;PersistentObjectTransactional(managerName = "fooManager")
057 *  public void doSomething() {
058 *      final Foo database = PersistentObjectTransactionManager.&lt;Foo&gt;getCurrent().getRoot();
059 *      ...
060 *  }
061 * </pre>
062 * </p>
063 *
064 * <p>
065 * To apply the AOP aspect associated with this annotation, the annoated method's class must be woven (either at build
066 * time or runtime) using the <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a>
067 * with the {@code PersistentObjectTransactionalAspect} aspect (included in the <code>dellroad-stuff</code> JAR file).
068 * <p>
069 *
070 * @see PersistentObjectTransactionManager
071 */
072@Documented
073@Inherited
074@Retention(RetentionPolicy.RUNTIME)
075@Target(ElementType.METHOD)
076public @interface PersistentObjectTransactional {
077
078    /**
079     * The {@linkplain PersistentObjectTransactionManager#getName name} of the desired
080     * {@link PersistentObjectTransactionManager}. This property only need be set when more than one
081     * {@link PersistentObjectTransactionManager} is configured with the aspect.
082     */
083    String managerName() default "";
084
085    /**
086     * Whether a read-only transaction should be performed; otherwise, the transaction is read-write.
087     */
088    boolean readOnly() default false;
089
090    /**
091     * Whether a read-only transaction should access the {@linkplain PersistentObject#getSharedRoot shared root}.
092     * In shared mode, copies are avoided entirely but the {@link PersistentObject} root object graph must not be
093     * modified, either by the annotated method, or by any caller to whom access is granted after the transaction
094     * ends by virtue of a returned reference.
095     *
096     * <p>
097     * If {@link #readOnly} is false, this property is ignored.
098     * </p>
099     */
100    boolean shared() default false;
101
102    /**
103     * Whether the transaction should execute asynchronously when in "Disruptor" mode.
104     *
105     * <p>
106     * Although all transactions are executed by the dedicated {@link PersistentObjectTransactionManager} transaction thread,
107     * by default {@link PersistentObjectTransactional @PersistentObjectTransactional}-annotated methods will block until the
108     * transaction completes, allowing the return value from the transaction to be passed back to the caller.
109     * </p>
110     *
111     * <p>
112     * Setting {@link #async} to true forces the method to return immediately, so that the transaction runs asynchronously.
113     * Such methods must return {@code void}. If an exception is thrown by an asynchronous transaction, it will be logged
114     * as an error and then discarded.
115     * </p>
116     *
117     * <p>
118     * Setting this property to true is only valid when the associated {@link PersistentObjectTransactionManager}
119     * is operating in "Disruptor" mode. Otherwise, an {@link IllegalStateException} is thrown by the annotated method.
120     * </p>
121     */
122    boolean async() default false;
123}
