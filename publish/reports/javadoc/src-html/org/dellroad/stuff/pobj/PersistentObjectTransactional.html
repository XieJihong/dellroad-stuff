008package org.dellroad.stuff.pobj;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Inherited;
013import java.lang.annotation.Retention;
014import java.lang.annotation.RetentionPolicy;
015import java.lang.annotation.Target;
016
017/**
018 * Declares that annotated methods should execute within {@link PersistentObjectTransactionManager} transactions.
019 *
020 * <p>
021 * When the associated AOP aspect is applied, annotated methods are executed on the {@link PersistentObjectTransactionManager}
022 * transaction thread, and the thread that invoked the method will block until the transaction completes (unless
023 * specified as {@link #async}). Return values and thrown exceptions are delivered back to the calling thread as
024 * they normally would be.
025 * </p>
026 *
027 * <p>
028 * The associated AOP aspect must be configured with the {@link PersistentObjectTransactionManager}(s) to be
029 * used for these transactional methods. The easiest way to do this is by declaring the aspect in your Spring
030 * bean factory, for example:
031 * <pre>
032 *  &lt;bean class="org.dellroad.stuff.pobj.PersistentObjectTransactionalAspect" factory-method="aspectOf"
033 *      p:persistentObjectTransactionManager-ref="myManagerBean"/&gt;
034 * </pre>
035 * </p>
036 *
037 * <p>
038 * More than one {@link PersistentObjectTransactionManager}s may be configured. For example:
039 * <pre>
040 *  &lt;bean class="org.dellroad.stuff.pobj.PersistentObjectTransactionalAspect" factory-method="aspectOf"&gt;
041 *      &lt;property name="persistentObjectTransactionManagers"&gt;
042 *          &lt;util:list&gt;
043 *              &lt;bean ref="myManagerBean1"/&gt;
044 *              &lt;bean ref="myManagerBean2"/&gt;
045 *          &lt;/util:list&gt;
046 *      &lt;/property&gt;
047 *  &lt;/bean&gt;
048 * </pre>
049 * In the case of multiple {@link PersistentObjectTransactionManager}s, which {@link PersistentObjectTransactionManager}
050 * the aspect uses is determined by the {@link #managerName} annotation property, which must equal the
051 * {@linkplain PersistentObjectTransactionManager#getName name} of the desired {@link PersistentObjectTransactionManager}, e.g.:
052 * <pre>
053
054 *  &#64;PersistentObjectTransactional(managerName = "myManagerBean1")
055 *  public void doSomething() {
056 *      ...
057 *  }
058 * </pre>
059 * </p>
060 *
061 * @see PersistentObjectTransactionManager
062 */
063@Documented
064@Inherited
065@Retention(RetentionPolicy.RUNTIME)
066@Target(ElementType.METHOD)
067public @interface PersistentObjectTransactional {
068
069    /**
070     * The {@linkplain PersistentObjectTransactionManager#getName name} of the desired
071     * {@link PersistentObjectTransactionManager}. This property only need be set when more than one
072     * {@link PersistentObjectTransactionManager} is configured with the aspect.
073     */
074    String managerName() default "";
075
076    /**
077     * Whether a read-only transaction should be performed; otherwise, the transaction is read-write.
078     */
079    boolean readOnly() default false;
080
081    /**
082     * Whether a read-only transaction should access the {@linkplain PersistentObject#getSharedRoot shared root}.
083     * In shared mode, copies are avoided entirely but the {@link PersistentObject} root object graph must not be
084     * modified, either by the annotated method, or by any caller to whom access is granted after the transaction
085     * ends by virtue of a returned reference.
086     *
087     * <p>
088     * If {@link #readOnly} is false, this property is ignored.
089     * </p>
090     */
091    boolean shared() default false;
092
093    /**
094     * Whether the transaction should execute asynchronously.
095     *
096     * <p>
097     * Although transactions always are executed by the dedicated {@link PersistentObjectTransactionManager} transaction thread,
098     * by default {@link PersistentObjectTransactional @PersistentObjectTransactional}-annotated methods will block until the
099     * transaction completes, allowing the return value from the transaction to be passed back to the caller.
100     * </p>
101     *
102     * <p>
103     * Setting {@link #async} to true forces the method to return immediately, so that the transaction runs asynchronously.
104     * Such methods must return {@code void}. If an exception is thrown by an asynchronous transaction, it will be logged
105     * as an error and then discarded.
106     * </p>
107     */
108    boolean async() default false;
109}
