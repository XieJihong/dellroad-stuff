008package org.dellroad.stuff.java;

010import java.lang.reflect.Method;
011import java.util.HashMap;
012import java.util.regex.Pattern;
013
014/**
015 * Simple utility enumeration for working Java {@link Class} instances representing primitive types.
016 *
017 * <p>
018 * Instances of this class represent one of the eight Java primitive types.
019 * <p/>
020 */
021public enum Primitive {
022
023    BOOLEAN(Boolean.TYPE, Boolean.class, 'Z', "true|false") {
024        @Override
025        public <R> R visit(PrimitiveSwitch<R> pswitch) {
026            return pswitch.caseBoolean();
027        }
028        @Override
029        public Object getDefaultValue() {
030            return false;
031        }
032    },
033    BYTE(Byte.TYPE, Byte.class, 'B', "(?i)(\\+|-)?(((0x|#)0*([A-F0-9]{1,2}))|(0+[0-7]{0,3})|([1-9][0-9]{0,2}))") {
034        @Override
035        public <R> R visit(PrimitiveSwitch<R> pswitch) {
036            return pswitch.caseByte();
037        }
038        @Override
039        public Object getDefaultValue() {
040            return (byte)0;
041        }
042    },
043    CHARACTER(Character.TYPE, Character.class, 'C', "(?s).") {
044        @Override
045        public <R> R visit(PrimitiveSwitch<R> pswitch) {
046            return pswitch.caseCharacter();
047        }
048        @Override
049        public Object getDefaultValue() {
050            return (char)0;
051        }
052    },
053    SHORT(Short.TYPE, Short.class, 'S', "(?i)(\\+|-)?(((0x|#)0*([A-F0-9]{1,4}))|(0+[0-7]{0,6})|([1-9][0-9]{0,4}))") {
054        @Override
055        public <R> R visit(PrimitiveSwitch<R> pswitch) {
056            return pswitch.caseShort();
057        }
058        @Override
059        public Object getDefaultValue() {
060            return (short)0;
061        }
062    },
063    INTEGER(Integer.TYPE, Integer.class, 'I', "(?i)(\\+|-)?(((0x|#)0*([A-F0-9]{1,8}))|(0+[0-7]{0,11})|([1-9][0-9]{0,9}))") {
064
065        @Override
066        public <R> R visit(PrimitiveSwitch<R> pswitch) {
067            return pswitch.caseInteger();
068        }
069        @Override
070        public Object getDefaultValue() {
071            return 0;
072        }
073    },
074    FLOAT(Float.TYPE, Float.class, 'F', DoubleFormat.REGEX) {
075        @Override
076        public <R> R visit(PrimitiveSwitch<R> pswitch) {
077            return pswitch.caseFloat();
078        }
079        @Override
080        public Object getDefaultValue() {
081            return (float)0;
082        }
083    },
084    LONG(Long.TYPE, Long.class, 'J', "(?i)(\\+|-)?(((0x|#)0*([A-F0-9]{1,16}))|(0+[0-7]{0,22})|([1-9][0-9]{0,18}))") {
085        @Override
086        public <R> R visit(PrimitiveSwitch<R> pswitch) {
087            return pswitch.caseLong();
088        }
089        @Override
090        public Object getDefaultValue() {
091            return (long)0;
092        }
093    },
094    DOUBLE(Double.TYPE, Double.class, 'D', DoubleFormat.REGEX) {
095        @Override
096        public <R> R visit(PrimitiveSwitch<R> pswitch) {
097            return pswitch.caseDouble();
098        }
099        @Override
100        public Object getDefaultValue() {
101            return (double)0;
102        }
103    };
104
105    private static final HashMap<Class<?>, Primitive> CLASS_MAP = new HashMap<Class<?>, Primitive>();
106
107    static {
108        for (Primitive prim : values()) {
109            CLASS_MAP.put(prim.primType, prim);
110            CLASS_MAP.put(prim.wrapType, prim);
111        }
112    }
113
114    private final Class<?> primType;
115    private final Class<?> wrapType;
116    private final char letter;
117    private final Pattern parsePattern;
118
119    Primitive(Class<?> primType, Class<?> wrapType, char letter, String parsePattern) {
120        this.primType = primType;
121        this.wrapType = wrapType;
122        this.letter = letter;
123        this.parsePattern = Pattern.compile(parsePattern);
124    }
125
126    public abstract <R> R visit(PrimitiveSwitch<R> pswitch);
127
128    public abstract Object getDefaultValue();
129
130    /**
131     * Get the short name for this primitive type, e.g., "int".
132     */
133    public String getName() {
134        return this.primType.getName();
135    }
136
137    /**
138     * Get the long name for this primitive type, e.g., "Integer".
139     */
140    public String getLongName() {
141        return this.wrapType.getSimpleName();
142    }
143
144    /**
145     * Get the single character descriptor for this primitive type, e.g., "I".
146     */
147    public char getLetter() {
148        return this.letter;
149    }
150
151    /**
152     * Get the {@link Class} object representing this primitive type, e.g., {@code Integer.TYPE}.
153     */
154    public Class<?> getType() {
155        return this.primType;
156    }
157
158    /**
159     * Get the wrapper {@link Class} object for this primitive type, e.g., {@code Integer.class}.
160     */
161    public Class<?> getWrapperType() {
162        return this.wrapType;
163    }
164
165    /**
166     * Get a regular expression that matches all allowed inputs to {@link #parseValue parseValue()}.
167     * The returned pattern may also accept some values that {@link #parseValue parseValue()} rejects,
168     * such as {@code 32768} for a {@code short}.
169     */
170    public Pattern getParsePattern() {
171        return this.parsePattern;
172    }
173
174    /**
175     * Get the wrapper class' "unwrap" method for this primitive type, e.g., {@code Integer.intValue()}.
176     */
177    public Method getUnwrapMethod() {
178        try {
179            return this.wrapType.getMethod(getName() + "Value");
180        } catch (NoSuchMethodException e) {
181            throw new RuntimeException(e);
182        }
183    }
184
185    /**
186     * Parse a string-encoded value of this instance.
187     * This method is guaranteed to accept all possible return values from the primitive's {@link #toString},
188     * and possibly other reasonable inputs (e.g., hex and octal values for integral types).
189     *
190     * @param string string representation of a value
191     * @throws IllegalArgumentException if {@code string} is null
192     * @throws IllegalArgumentException if {@code string} does not match the {@linkplain #getParsePattern parse pattern}
193     */
194    public Object parseValue(String string) {
195
196        // Handle null
197        if (string == null)
198            throw new IllegalArgumentException("null string");
199
200        // Parse value
201        switch (this) {
202        case BOOLEAN:
203            return Boolean.parseBoolean(string);
204        case BYTE:
205            return Byte.decode(string);
206        case CHARACTER:
207            if (string.length() != 1)
208                throw new IllegalArgumentException("string has length " + string.length() + " != 1");
209            return Character.valueOf(string.charAt(0));
210        case SHORT:
211            return Short.decode(string);
212        case INTEGER:
213            return Integer.decode(string);
214        case FLOAT:
215            return Float.parseFloat(string);
216        case LONG:
217            return Long.decode(string);
218        case DOUBLE:
219            return Double.parseDouble(string);
220        default:
221            throw new RuntimeException();
222        }
223    }
224
225    /**
226     * Get the value corresponding to the given Java primitive or primitive wrapper type.
227     *
228     * @return the {@link Primitive} corresponding to {@code c}, or {@code null}
229     *         if {@code c} is not a primitive or primitive wrapper type
230     */
231    public static Primitive get(Class<?> c) {
232        return CLASS_MAP.get(c);
233    }

235    // This is put into an inner class to avoid initialization ordering problems
236    private static final class DoubleFormat {
237
238        private static final String DIGITS = "(\\p{Digit}+)";
239
240        private static final String HEX_DIGITS = "(\\p{XDigit}+)";
241
242        // an exponent is 'e' or 'E' followed by an optionally
243        // signed decimal integer.
244        private static final String EXPONENT = "[eE][+-]?" + DIGITS;
245
246        private static final String REGEX =
247          "[+-]?("              // Optional sign character
248          + "NaN|"              // "NaN" string
249          + "Infinity|"         // "Infinity" string
250
251          // A decimal floating-point string representing a finite positive
252          // number without a leading sign has at most five basic pieces:
253          // Digits . Digits ExponentPart FloatTypeSuffix
254          //
255          // Since this method allows integer-only strings as input
256          // in addition to strings of floating-point literals, the
257          // two sub-patterns below are simplifications of the grammar
258          // productions from the Java Language Specification, 2nd
259          // edition, section 3.10.2.
260
261          // Digits ._opt Digits_opt ExponentPart_opt FloatTypeSuffix_opt
262          + "(((" + DIGITS + "(\\.)?(" + DIGITS + "?)(" + EXPONENT + ")?)|"
263
264          // . Digits ExponentPart_opt FloatTypeSuffix_opt
265          + "(\\.(" + DIGITS + ")(" + EXPONENT + ")?)|"
266
267          // Hexadecimal strings
268          + "(("
269            // 0[xX] HexDigits ._opt BinaryExponent FloatTypeSuffix_opt
270            + "(0[xX]" + HEX_DIGITS + "(\\.)?)|"
271
272            // 0[xX] HexDigits_opt . HexDigits BinaryExponent FloatTypeSuffix_opt
273            + "(0[xX]" + HEX_DIGITS + "?(\\.)" + HEX_DIGITS + ")"
274             + ")[pP][+-]?" + DIGITS + "))" + "[fFdD]?))";
275
276        private DoubleFormat() {
277        }
278    }
