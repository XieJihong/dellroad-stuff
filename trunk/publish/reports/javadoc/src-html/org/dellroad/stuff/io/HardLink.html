008package org.dellroad.stuff.io;
009
010import com.sun.jna.Library;
011import com.sun.jna.Native;
012
013import java.io.File;
014import java.io.IOException;
015
016/**
017 * Supports hard-linking of files.
018 *
019 * <p>
020 * On Java &gt;= 1.7, this class uses {@link java.nio.file.Files#createLink}.
021 * On Java &lt; 1.7, this class requires the <a href="https://github.com/twall/jna">JNA</a> library
022 * and only works on UNIX systems.
023 * </p>
024 */
025public final class HardLink {
026
027    private static LibC libc;
028
029    private HardLink() {
030    }
031
032    /**
033     * Create a hard link from {@code src} to {@code dest}.
034     *
035     * @param src existing file to be linked to {@code dest}
036     * @param dest new file that will link to {@code src}
037     * @throws IOException if the operation fails
038     */
039    public static void link(File src, File dest) throws IOException {
040
041        // Try JDK 1.7 Files.createLink() first
042        try {
043            Class.forName("java.nio.files.Files").getMethod("createLink").invoke(null, dest, src);
044            return;
045        } catch (Exception e) {
046            // ignore
047        }
048
049        // Fall-back to JNA
050        if (HardLink.getLibC().link(src.toString(), dest.toString()) != 0)
051            throw new IOException(HardLink.getLibC().strerror(Native.getLastError()));
052    }
053
054    private static LibC getLibC() {
055        if (HardLink.libc == null)
056            HardLink.libc = (LibC)Native.loadLibrary("c", LibC.class);
057        return HardLink.libc;
058    }
059
060    /**
061     * Command line test method.
062     */
063    static void main(String[] args) throws Exception {
064        HardLink.link(new File(args[0]), new File(args[1]));
065    }
066
067    private interface LibC extends Library {
068        int link(String from, String to);
069        String strerror(int errno);
070    }
071}
