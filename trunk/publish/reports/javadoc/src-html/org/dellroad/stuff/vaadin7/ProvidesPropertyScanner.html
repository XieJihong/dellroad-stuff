008package org.dellroad.stuff.vaadin7;
009
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.HashMap;
013import java.util.List;
014
015import org.dellroad.stuff.java.MethodAnnotationScanner;
016import org.dellroad.stuff.java.Primitive;
017
018/**
019 * Scans a Java class hierarchy for {@link ProvidesProperty &#64;ProvidesProperty} and
020 * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated getter methods and creates
021 * a corresponding set of {@link PropertyDef} property definitions and a {@link SortingPropertyExtractor}
022 * that will extract the properties from instances of the given class and sort them accordingly.
023 *
024 * @param <T> Java class to be introspected
025 * @see ProvidesProperty &#64;ProvidesProperty
026 * @see ProvidesPropertySort &#64;ProvidesPropertySort
027 */
028public class ProvidesPropertyScanner<T> {
029
030    private final ArrayList<PropertyDef<?>> propertyList = new ArrayList<PropertyDef<?>>();
031    private final SortingPropertyExtractor<T> propertyExtractor;
032
033    /**
034     * Constructor.
035     *
036     * @param type Java class to be introspected
037     * @throws IllegalArgumentException if {@code type} is null
038     * @throws IllegalArgumentException if an annotated method with no {@linkplain ProvidesProperty#value property name specified}
039     *  has a name which cannot be interpreted as a bean property "getter" method
040     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
041     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
042     */
043    public ProvidesPropertyScanner(Class<T> type) {
044
045        // Sanity check
046        if (type == null)
047            throw new IllegalArgumentException("null type");
048
049        // Scan for @ProvidesProperty and @ProvidesPropertySort annotations
050        final List<MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyMethods
051          = new MethodAnnotationScanner<T, ProvidesProperty>(type, ProvidesProperty.class).buildMethodInfoList();
052        final List<MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortMethods
053          = new MethodAnnotationScanner<T, ProvidesPropertySort>(type, ProvidesPropertySort.class).buildMethodInfoList();
054
055        // Check for duplicate @ProvidesProperty names
056        final HashMap<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyNameMap
057          = new HashMap<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo>();
058        for (MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo : providesPropertyMethods) {
059            final String propertyName = this.getPropertyName(methodInfo);
060            final MethodAnnotationScanner<T, ?>.MethodInfo previous = providesPropertyNameMap.put(propertyName, methodInfo);
061
062            // Check for name conflict
063            if (previous != null) {
064                throw new IllegalArgumentException("duplicate @" + ProvidesProperty.class.getSimpleName()
065                  + " declaration for property `" + propertyName + "' on method " + previous.getMethod()
066                  + " and " + methodInfo.getMethod());
067            }
068        }
069
070        // Check for duplicate @ProvidesPropertySort names, etc.
071        final HashMap<String, MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortNameMap
072          = new HashMap<String, MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo>();
073        for (MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo : providesPropertySortMethods) {
074            final String propertyName = this.getSortPropertyName(methodInfo);
075            final MethodAnnotationScanner<T, ?>.MethodInfo previous = providesPropertySortNameMap.put(propertyName, methodInfo);
076
077            // Check for name conflict
078            if (previous != null) {
079                throw new IllegalArgumentException("duplicate @" + ProvidesPropertySort.class.getSimpleName()
080                  + " declaration for property `" + propertyName + "' on method " + previous.getMethod()
081                  + " and " + methodInfo.getMethod());
082            }
083
084            // Get method type; if primitive, get wrapper type
085            Class<?> methodType = methodInfo.getMethod().getReturnType();
086            if (methodType.isPrimitive())
087                methodType = Primitive.get(methodType).getWrapperType();
088
089            // Verify type is Comparable
090            if (!Comparable.class.isAssignableFrom(methodType)) {
091                throw new IllegalArgumentException("invalid @" + ProvidesPropertySort.class.getSimpleName()
092                  + " declaration for property `" + propertyName + "': method " + methodInfo.getMethod()
093                  + " return type " + methodType.getName() + " is not an instance of " + Comparable.class.getName());
094            }
095        }
096
097        // Build PropertyDef list
098        for (MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo : providesPropertyMethods) {
099            final String propertyName = this.getPropertyName(methodInfo);
100
101            // Get property type
102            Class<?> propertyType = methodInfo.getMethod().getReturnType();
103
104            // Get property default value
105            Object defaultValue = null;
106            if (propertyType.isPrimitive()) {
107                final Primitive primitiveType = Primitive.get(propertyType);
108                defaultValue = primitiveType.getDefaultValue();
109                propertyType = primitiveType.getWrapperType();
110            }
111
112            // Get sort property (if any)
113            final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo
114              = providesPropertySortNameMap.get(propertyName);
115
116            // Create property definition
117            this.propertyList.add(this.createAnnotationPropertyDef(
118              propertyName, propertyType, defaultValue, methodInfo, sortMethodInfo));
119        }
120
121        // Build PropertyExtractor
122        this.propertyExtractor = new SortingPropertyExtractor<T>() {
123
124            @Override
125            @SuppressWarnings("unchecked")
126            public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
127                if (!(propertyDef instanceof AnnotationPropertyDef))
128                    throw new RuntimeException("unknown property " + propertyDef);
129                final AnnotationPropertyDef<V> annotationPropertyDef = (AnnotationPropertyDef<V>)propertyDef;
130                return propertyDef.getType().cast(annotationPropertyDef.getMethodInfo().readValue(obj));
131            }
132
133            @Override
134            @SuppressWarnings("unchecked")
135            public boolean canSort(PropertyDef<?> propertyDef) {
136                if (!(propertyDef instanceof AnnotationPropertyDef))
137                    return false;
138                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef)propertyDef;
139                return annotationPropertyDef.getSortMethodInfo() != null;
140            }
141
142            @Override
143            @SuppressWarnings("unchecked")
144            public int sort(PropertyDef<?> propertyDef, T obj1, T obj2) {
145                if (!(propertyDef instanceof AnnotationPropertyDef))
146                    throw new UnsupportedOperationException("unknown property " + propertyDef);
147                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef)propertyDef;
148                final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo
149                  = annotationPropertyDef.getSortMethodInfo();
150                if (sortMethodInfo == null)
151                    throw new UnsupportedOperationException("can't sort property " + propertyDef);
152                final Comparable<Object> value1 = (Comparable<Object>)annotationPropertyDef.getSortMethodInfo().readValue(obj1);
153                final Comparable<Object> value2 = (Comparable<Object>)annotationPropertyDef.getSortMethodInfo().readValue(obj2);
154                if (value1 == null && value2 != null)
155                    return -1;
156                if (value1 != null && value2 == null)
157                    return 1;
158                if (value1 == null && value2 == null)
159                    return 0;
160                return value1.compareTo(value2);
161            }
162        };
163    }
164
165    /**
166     * Get the list of {@link PropertyDef}s generated from the annotated methods.
167     *
168     * <p>
169     * All of the properties in the returned list can be extracted from instances of this reader's configured
170     * class by the {@link PropertyExtractor} returned by {@link #getPropertyExtractor}.
171     * </p>
172     *
173     * @return unmodifiable list of properties
174     * @see #getPropertyExtractor
175     */
176    public List<PropertyDef<?>> getPropertyDefs() {
177        return Collections.unmodifiableList(this.propertyList);
178    }
179
180    /**
181     * Get the {@link PropertyExtractor} that extracts Vaadin {@link com.vaadin.data.Property} values
182     * from instances of the annotated class when given one of the {@link PropertyDef}s returned by
183     * {@link #getPropertyDefs}.
184     *
185     * @see #getPropertyDefs
186     */
187    public SortingPropertyExtractor<T> getPropertyExtractor() {
188        return this.propertyExtractor;
189    }
190
191    private String getPropertyName(MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo) {
192        return methodInfo.getAnnotation().value().length() > 0 ?
193          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
194    }
195
196    private String getSortPropertyName(MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo) {
197        return methodInfo.getAnnotation().value().length() > 0 ?
198          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
199    }
200
201    // This method exists solely to bind the generic type
202    private <V> AnnotationPropertyDef<V> createAnnotationPropertyDef(String propertyName, Class<V> propertyType,
203      Object defaultValue, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
204      MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
205        return new AnnotationPropertyDef<V>(propertyName, propertyType,
206          propertyType.cast(defaultValue), methodInfo, sortMethodInfo);
207    }
208
209    private class AnnotationPropertyDef<V> extends PropertyDef<V> {
210
211        private final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo;
212        private final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo;
213
214        AnnotationPropertyDef(String name, Class<V> type, V defaultValue,
215          MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
216          MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
217            super(name, type, defaultValue);
218            this.methodInfo = methodInfo;
219            this.sortMethodInfo = sortMethodInfo;
220        }
221
222        public MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo getMethodInfo() {
223            return this.methodInfo;
224        }
225
226        public MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo getSortMethodInfo() {
227            return this.sortMethodInfo;
228        }
229    };
230}
