026/**
027 * {@link PersistentObject} implementation of Spring's
028 * {@link org.springframework.transaction.PlatformTransactionManager PlatformTransactionManager} interface,
029 * allowing methods annotated with Spring's {@link org.springframework.transaction.annotation.Transactional &#64;Transactional}
030 * annotation to perform transactions on {@link PersistentObject}s.
031 *
032 * <p>
033 * During a transaction, the transaction's root object is available via the method {@link #getRoot}, and may be replaced entirely
034 * via {@link #setRoot setRoot()}. When the transaction completes, the transaction's root object will be automatically
035 * written back to the {@link PersistentObject} via {@link PersistentObject#setRoot(Object, long) PersistentObject.setRoot()}
036 * (unless the transaction was read-only). During commit, the {@link PersistentObject} version number is verified, and
037 * if another update has occurred since the transaction was opened, a {@link PersistentObjectVersionException} is thrown
038 * (consider using {@link org.dellroad.stuff.spring.RetryTransaction &#64;RetryTransaction} for automatic retry in this case).
039 * </p>
040 *
041 * <p>
042 * The {@code persistentObject} property is required. The {@code readOnlySharedRoot} property is optional and configures whether,
043 * during read-only transactions only, {@link #getRoot} returns the {@linkplain PersistentObject#getSharedRoot shared root} object.
044 * In this mode, root object graph copies are avoided entirely for read-only transactions, but the application code must be
045 * careful to not modify the object graph returned by {@link #getRoot} at any time, either during or after the transaction.
046 * </p>
047 *
048 * @param <T> type of the root object
049 * @see PersistentObject
050 * @see PersistentObject#getSharedRoot
051 */
052@SuppressWarnings("serial")
053public class PersistentObjectTransactionManager<T> extends AbstractPlatformTransactionManager
054  implements BeanNameAware, ResourceTransactionManager, InitializingBean {
055
056    private static final ThreadLocal<HashMap<String, PersistentObjectTransactionManager<?>>> MANAGER_MAP = new ThreadLocal<>();
057
058    /**
059     * The configured {@link PersistentObject} from which transactions are created.
060     */
061    protected transient PersistentObject<T> persistentObject;
062
063    private final ThreadLocal<TxInfo<T>> currentInfo = new ThreadLocal<>();
064
065    private String beanName;
066    private boolean readOnlySharedRoot;

110// Current Instance

112    /**
113     * Get the instance associated with an open transaction in the current thread and having the given bean name.
114     *
115     * @param beanName bean name assigned to the desired instance
116     * @throws IllegalArgumentException if {@code name} is null
117     * @throws IllegalStateException if the current thread is not running within a transaction managed by
118     *  a {@link PersistentObjectTransactionManager} assigned the given name
119     */
120    @SuppressWarnings("unchecked")
121    public static <T> PersistentObjectTransactionManager<T> getCurrent(String beanName) {
122        if (beanName == null)
123            throw new IllegalArgumentException("null beanName");
124        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
125          = PersistentObjectTransactionManager.getManagerMap(false);
126        final PersistentObjectTransactionManager<?> manager = managerMap != null ? managerMap.get(beanName) : null;
127        if (manager == null) {
128            throw new IllegalStateException("no PersistentObjectTransactionManager named `" + beanName
129              + "' has an open transaction in the current thread");
130        }
131        return (PersistentObjectTransactionManager<T>)manager;
132    }
133
134    /**
135     * Get the (unique) instance associated with an open transaction in the current thread.
136     *
137     * <p>
138     * This is a convenience method for the common case where there is only one instance associated with the current thread.
139     * </p>
140     *
141     * @throws IllegalStateException if the current thread is not running within a
142     *  {@link PersistentObjectTransactionManager} transaction
143     * @throws IllegalStateException there is more than one {@link PersistentObjectTransactionManager} transaction
144     *  open in the current thread
145     */
146    @SuppressWarnings("unchecked")
147    public static <T> PersistentObjectTransactionManager<T> getCurrent() {
148        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
149          = PersistentObjectTransactionManager.getManagerMap(false);
150        if (managerMap == null || managerMap.isEmpty()) {
151            throw new IllegalStateException("there are no PersistentObjectTransactionManager transactions open"
152              + " in the current thread");
153        }
154        if (managerMap.size() > 1) {
155            throw new IllegalStateException("there are multiple PersistentObjectTransactionManager transactions open"
156              + " in the current thread: " + managerMap.keySet() + "; invoke getCurrent() with explicitly specified bean name");
157        }
158        return (PersistentObjectTransactionManager<T>)managerMap.values().iterator().next();
159    }
160
161    private static HashMap<String, PersistentObjectTransactionManager<?>> getManagerMap(boolean create) {
162        HashMap<String, PersistentObjectTransactionManager<?>> managerMap = MANAGER_MAP.get();
163        if (managerMap == null && create) {
164            managerMap = new HashMap<String, PersistentObjectTransactionManager<?>>();
165            MANAGER_MAP.set(managerMap);
166        }
167        return managerMap;
168    }

170// Root Object Access
171
172    /**
173     * Determine whether the transaction associated with the current thread is a read-only transaction.
174     *
175     * @return true if the transaction associated with the current thread is a read-only transaction
176     * @throws IllegalStateException if there is no transaction associated with the current thread
177     */
178    public boolean isReadOnly() {
179        final TxInfo<T> info = this.currentInfo.get();
180        if (info == null)
181            throw new IllegalStateException("there is no transaction associated with the current thread");
182        return info.isReadOnly();
183    }
184
185    /**
186     * Get the snapshot version of the transaction associated with the current thread.
187     *
188     * @return snapshot version of the transaction associated with the current thread.
189     * @throws IllegalStateException if there is no transaction associated with the current thread
190     */
191    public long getSnapshotVersion() {
192        final TxInfo<T> info = this.currentInfo.get();
193        if (info == null)
194            throw new IllegalStateException("there is no transaction associated with the current thread");
195        return info.getSnapshot().getVersion();
196    }
197
198    /**
199     * Get the root object graph for use in the transaction associated with the current thread.
200     *
201     * @return root object
202     * @throws IllegalStateException if there is no open transaction
203     */
204    public T getRoot() {
205        final TxInfo<T> info = this.currentInfo.get();
206        if (info == null)
207            throw new IllegalStateException("there is no transaction associated with the current thread");
208        return info.getSnapshot().getRoot();
209    }
210
211    /**
212     * Change the root object graph to be committed in the transaction associated with the current thread.
213     * Subsequent calls to {@link #getRoot getRoot} will return the new object.
214     *
215     * <p>
216     * This method may be invoked during read-only transactions (as before, the root will not actually be committed).
217     * </p>
218     *
219     * @param root new root object
220     * @throws IllegalStateException if there is no transaction associated with the current thread
221     * @throws IllegalArgumentException if {@code root} is null
222     */
223    public void setRoot(T root) {
224        if (root == null)
225            throw new IllegalArgumentException("null root");
226        final TxInfo<T> info = this.currentInfo.get();
227        if (info == null)
228            throw new IllegalStateException("there is no transaction associated with the current thread");
229        info.setSnapshot(this.persistentObject.new Snapshot(root, info.getSnapshot().getVersion()));
230    }

241    @Override
242    protected Object doGetTransaction() {
243        return new TxWrapper<T>(this.currentInfo.get());
244    }

246    @Override
247    protected boolean isExistingTransaction(Object txObj) {
248        return ((TxWrapper<?>)txObj).getInfo() != null;
249    }
250
251    @Override
252    @SuppressWarnings("unchecked")
253    protected void doBegin(Object txObj, TransactionDefinition txDef) {
254
255        // Sanity check
256        if (this.currentInfo.get() != null)
257            throw new TransactionUsageException("there is already a transaction associated with the current thread");
258        if (this.isExistingTransaction(txObj))
259            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
260        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
261
262        // Create transaction
263        final PersistentObject<T>.Snapshot snapshot;
264        try {
265            snapshot = txDef.isReadOnly() && this.readOnlySharedRoot ?
266              this.persistentObject.getSharedRootSnapshot() : this.persistentObject.getRootSnapshot();
267        } catch (PersistentObjectException e) {
268            throw new CannotCreateTransactionException("error creating new JSimpleDB transaction", e);
269        }
270
271        // Associate transaction with this thread for getCurrent()
272        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
273          = PersistentObjectTransactionManager.getManagerMap(true);
274        if (managerMap.containsKey(this.beanName)) {
275            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
276              + "' already has an open transaction in the current thread; all bean names must be distinct");
277        }
278        managerMap.put(this.beanName, this);
279
280        // Associate new transation with the current thread
281        final TxInfo<T> info = new TxInfo<T>(snapshot, txDef.isReadOnly());
282        this.currentInfo.set(info);
283
284        // Set transaction info into the Spring transaction object
285        tx.setInfo(info);
286    }
287
288    /**
289     * Suspend the current transaction.
290     */
291    @Override
292    @SuppressWarnings("unchecked")
293    protected Object doSuspend(Object txObj) {
294
295        // Sanity check
296        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
297        final TxInfo<T> info = tx.getInfo();
298        if (info == null)
299            throw new TransactionUsageException("no PersistentObject transaction exists in the given transaction object");
300        if (info != this.currentInfo.get())
301            throw new TransactionUsageException("the provided transaction object contains the wrong PersistentObject transaction");
302
303        // Suspend it
304        if (this.logger.isTraceEnabled())
305            this.logger.trace("suspending current PersistentObject transaction" + info.getSnapshot());
306        this.currentInfo.remove();
307        tx.setInfo(null);
308        PersistentObjectTransactionManager.getManagerMap(true).remove(this.beanName);
309
310        // Done
311        return info;
312    }
313
314    /**
315     * Resume a previously suspended transaction.
316     */
317    @Override
318    @SuppressWarnings("unchecked")
319    protected void doResume(Object txObj, Object suspendedResources) {
320
321        // Sanity check
322        if (this.currentInfo.get() != null)
323            throw new TransactionUsageException("there is already a transaction associated with the current thread");
324        if (this.isExistingTransaction(txObj))
325            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
326
327        // Get previously saved info
328        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
329        final TxInfo<T> info = (TxInfo<T>)suspendedResources;
330
331        // Resume transaction
332        if (this.logger.isTraceEnabled())
333            this.logger.trace("resuming PersistentObject transaction " + info.getSnapshot());
334        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
335          = PersistentObjectTransactionManager.getManagerMap(true);
336        if (managerMap.containsKey(this.beanName)) {
337            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
338              + "' already has an open transaction in the current thread; all bean names must be distinct");
339        }
340        managerMap.put(this.beanName, this);
341        this.currentInfo.set(info);
342        tx.setInfo(info);
343    }

345    @Override
346    @SuppressWarnings("unchecked")
347    protected void prepareForCommit(DefaultTransactionStatus status) {
348        final TxInfo<T> info = ((TxWrapper<T>)status.getTransaction()).getInfo();
349        if (info == null)
350            throw new NoTransactionException("no current transaction exists");
351    }
352
353    @Override
354    @SuppressWarnings("unchecked")
355    protected void doCommit(DefaultTransactionStatus status) {
356
357        // Get transaction
358        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
359        final TxInfo<T> info = tx.getInfo();
360        if (info == null)
361            throw new NoTransactionException("no current transaction exists");
362        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
363
364        // Commit
365        try {
366            if (!info.isReadOnly()) {
367                if (this.logger.isTraceEnabled())
368                    this.logger.trace("committing PersistentObject transaction " + snapshot);
369                this.persistentObject.setRoot(snapshot.getRoot(), snapshot.getVersion());
370            } else
371                this.logger.trace("not committing read-only PersistentObject transaction " + snapshot);
372        } catch (PersistentObjectVersionException e) {
373            throw new OptimisticLockingFailureException(null, e);
374        } catch (PersistentObjectValidationException e) {
375            throw new DataIntegrityViolationException(null, e);
376        } catch (PersistentObjectException e) {
377            throw new TransactionSystemException("error committing transaction", e);
378        }
379    }
380
381    @Override
382    @SuppressWarnings("unchecked")
383    protected void doRollback(DefaultTransactionStatus status) {
384
385        // Get transaction
386        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
387        final TxInfo<T> info = tx.getInfo();
388        if (info == null)
389            throw new NoTransactionException("no current transaction exists");
390        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
391
392        // Rollback
393        if (this.logger.isTraceEnabled())
394            this.logger.trace("rolling back PersistentObject transaction " + snapshot);
395        // no action required to rollback a PersistentObject transaction
396    }
397
398    @Override
399    @SuppressWarnings("unchecked")
400    protected void doSetRollbackOnly(DefaultTransactionStatus status) {
401
402        // Get transaction
403        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
404        final TxInfo<T> info = tx.getInfo();
405        if (info == null)
406            throw new NoTransactionException("no current transaction exists");
407
408        // Set rollback only
409        if (this.logger.isTraceEnabled())
410            this.logger.trace("marking PersistentObject transaction " + info.getSnapshot() + " for rollback-only");
411        tx.setRollbackOnly(true);
412    }
413
414    @Override
415    @SuppressWarnings("unchecked")
416    protected void doCleanupAfterCompletion(Object txObj) {
417
418        // Get transaction
419        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
420
421        // Clean up
422        tx.setInfo(null);
423        this.currentInfo.remove();
424        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
425          = PersistentObjectTransactionManager.getManagerMap(false);
426        if (managerMap != null) {
427            managerMap.remove(this.beanName);
428            if (managerMap.isEmpty())
429                MANAGER_MAP.remove();
430        }
431    }

435    // Information passed to/from superclass about a transaction
436    private static class TxWrapper<S> implements SmartTransactionObject {
437
438        private TxInfo<S> info;
439        private boolean rollbackOnly;
440
441        TxWrapper(TxInfo<S> info) {
442            this.info = info;
443        }
444
445        public TxInfo<S> getInfo() {
446            return this.info;
447        }
448        public void setInfo(TxInfo<S> info) {
449            this.info = info;
450        }
451
452        @Override
453        public boolean isRollbackOnly() {
454            return this.rollbackOnly;
455        }
456        public void setRollbackOnly(boolean rollbackOnly) {
457            this.rollbackOnly = rollbackOnly;
458        }
459
460        @Override
461        public void flush() {
462        }
463    }

465// TxInfo

467    // Information about the transaction associated with the current thread
468    private static class TxInfo<S> {
469
470        private PersistentObject<S>.Snapshot snapshot;
471        private final boolean readOnly;
472
473        TxInfo(PersistentObject<S>.Snapshot snapshot, boolean readOnly) {
474            this.setSnapshot(snapshot);
475            this.readOnly = readOnly;
476        }
477
478        public PersistentObject<S>.Snapshot getSnapshot() {
479            return this.snapshot;
480        }
481        public void setSnapshot(PersistentObject<S>.Snapshot snapshot) {
482            if (snapshot == null)
483                throw new IllegalArgumentException("null snapshot");
484            this.snapshot = snapshot;
485        }
486
487        public boolean isReadOnly() {
488            return this.readOnly;
489        }
490    }
