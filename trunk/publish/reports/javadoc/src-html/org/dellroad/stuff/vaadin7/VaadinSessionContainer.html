008package org.dellroad.stuff.vaadin7;
009
010import com.vaadin.server.SessionDestroyEvent;
011import com.vaadin.server.SessionDestroyListener;
012import com.vaadin.server.SessionInitEvent;
013import com.vaadin.server.SessionInitListener;
014import com.vaadin.server.VaadinService;
015import com.vaadin.server.VaadinServlet;
016import com.vaadin.server.VaadinServletService;
017import com.vaadin.server.VaadinSession;
018
019import java.util.ArrayList;
020import java.util.Collection;
021
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * Container containing active {@link VaadinSession}s. This class is useful when you need to display
027 * sessions in your GUI, for example, a table showing all logged-in users.
028 *
029 * <p>
030 * By subclassing this class and {@link VaadinSessionInfo}, additional session-related properties,
031 * such as logged-in user, can be added to the container by annotating the {@link VaadinSessionInfo}
032 * subclass with {@link ProvidesProperty &#64;ProvidesProperty} annotations.
033 * </p>
034 *
035 * <p>
036 * Note the following:
037 * <ul>
038 *  <li>The {@link SpringVaadinServlet} must be used and configured with init parameter
039 *      {@link SpringVaadinServlet#SESSION_TRACKING_PARAMETER} set to {@code true}.</li>
040 *  <li>The {@link #connect} and {@link #disconnect} methods must be invoked before and after (respectively)
041 *      this container is used; typically these would be invoked in the {@link com.vaadin.ui.Component#attach}
042 *      and {@link com.vaadin.ui.Component#detach} methods of a corresponding widget.
043 * </ul>
044 * </p>
045 *
046 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
047 */
048@SuppressWarnings("serial")
049public abstract class VaadinSessionContainer<T extends VaadinSessionInfo> extends SimpleKeyedContainer<VaadinSession, T> {
050
051    /**
052     * The {@link VaadinSession} that this container instance is associated with.
053     * Determined by whatever {@link VaadinSession} is associated with the current thread at construction time.
054     */
055    protected final VaadinSession session;
056
057    private final Logger log = LoggerFactory.getLogger(this.getClass());
058    private final SessionEventListener listener = new SessionEventListener();
059
060    /**
061     * Constructor.
062     *
063     * <p>
064     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
065     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
066     * </p>
067     *
068     * @param type class to introspect for annotated methods
069     * @throws IllegalArgumentException if {@code type} is null
070     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
071     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
072     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
073     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
074     *  property "getter" method
075     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
076     * @throws IllegalStateException if there is no {@link SpringVaadinServlet} associated
077     *  with the current thread's {@link VaadinSession}
078     * @see ProvidesProperty
079     * @see ProvidesPropertySort
080     * @see ProvidesPropertyScanner
081     */
082    protected VaadinSessionContainer(Class<T> type) {
083        super(type);
084        this.session = VaadinUtil.getCurrentSession();
085        this.getSpringVaadinServlet();
086    }
087
088    /**
089     * Constructor.
090     *
091     * @param propertyExtractor used to extract properties from the underlying Java objects;
092     *  may be null but then container is not usable until one is configured via
093     * {@link #setPropertyExtractor setPropertyExtractor()}
094     * @param propertyDefs container property definitions; null is treated like the empty set
095     */
096    protected VaadinSessionContainer(PropertyExtractor<? super T> propertyExtractor,
097      Collection<? extends PropertyDef<?>> propertyDefs) {
098        super(propertyExtractor, propertyDefs);
099        this.session = VaadinUtil.getCurrentSession();
100        this.getSpringVaadinServlet();
101    }
102
103    @Override
104    protected VaadinSession getKeyFor(T info) {
105        return info.getVaadinSession();
106    }
107
108    /**
109     * Start tracking live sessions.
110     *
111     */
112    public void connect() {
113        this.listener.register();
114    }
115
116    /**
117     * Stop tracking live sessions.
118     */
119    public void disconnect() {
120        this.listener.unregister();
121    }
122
123    /**
124     * Reload this container.
125     *
126     * <p>
127     * This method can be invoked from any thread. It creates a new thread to do the actual
128     * reloading to avoid potential deadlocks.
129     * </p>
130     */
131    public void reload() {
132        new Thread("VaadinSessionContainer.reload()") {
133            @Override
134            public void run() {
135                try {
136                    VaadinSessionContainer.this.update();
137                } catch (ThreadDeath t) {
138                    throw t;
139                } catch (Throwable t) {
140                    VaadinSessionContainer.this.log.error("error updating container " + this, t);
141                }
142            }
143        }.start();
144    }
145
146    /**
147     * Update this container after a {@link VaadinSession} has been added or removed.
148     * This must <b>not</b> be invoked while any {@link VaadinSession} is locked.
149     */
150    protected void update() {
151
152        // Sanity check
153        if (VaadinSession.getCurrent() != null && VaadinSession.getCurrent().hasLock())
154            throw new IllegalStateException("inside locked VaadinSession");
155
156        // Create a VaadinSessionInfo object for each session, but doing so while that session is locked
157        final ArrayList<T> sessionInfoList = new ArrayList<T>();
158        for (VaadinSession otherSession : this.getSpringVaadinServlet().getSessions()) {
159            otherSession.accessSynchronously(new Runnable() {
160                @Override
161                public void run() {
162                    sessionInfoList.add(VaadinSessionContainer.this.createVaadinSessionInfo());
163                }
164            });
165        }
166
167        // Reload this container
168        this.session.access(new Runnable() {
169            @Override
170            public void run() {
171                VaadinSessionContainer.this.load(sessionInfoList);
172            }
173        });
174    }
175
176    protected SpringVaadinServlet getSpringVaadinServlet() {
177        final VaadinService service = this.session.getService();
178        if (!(service instanceof VaadinServletService))
179            throw new IllegalStateException("there is no SpringVaadinServlet associated with this VaadinSession");
180        final VaadinServlet servlet = ((VaadinServletService)service).getServlet();
181        if (!(servlet instanceof SpringVaadinServlet))
182            throw new IllegalStateException("there is no SpringVaadinServlet associated with this VaadinSession");
183        return (SpringVaadinServlet)servlet;
184    }
185
186    /**
187     * Create a {@link VaadinSessionInfo} backing object for the {@link VaadinSession} associated with the current thread.
188     * The {@link VaadinSession} will be already locked.
189     */
190    protected abstract T createVaadinSessionInfo();
191
192// SessionEventListener
193
194    private class SessionEventListener extends VaadinExternalListener<VaadinService>
195      implements SessionInitListener, SessionDestroyListener {
196
197        public SessionEventListener() {
198            super(VaadinUtil.getCurrentSession().getService());
199        }
200
201        @Override
202        public void sessionInit(SessionInitEvent event) {
203            VaadinSessionContainer.this.reload();
204        }
205
206        @Override
207        public void sessionDestroy(SessionDestroyEvent event) {
208            VaadinSessionContainer.this.reload();
209        }
210
211        @Override
212        protected void register(VaadinService vaadinService) {
213            vaadinService.addSessionInitListener(this);
214            vaadinService.addSessionDestroyListener(this);
215        }
216
217        @Override
218        protected void unregister(VaadinService vaadinService) {
219            vaadinService.removeSessionInitListener(this);
220            vaadinService.removeSessionDestroyListener(this);
221        }
222    }
223}
