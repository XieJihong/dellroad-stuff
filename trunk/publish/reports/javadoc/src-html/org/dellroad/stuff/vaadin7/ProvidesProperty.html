008package org.dellroad.stuff.vaadin7;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotates a Java field or method that provides a read-only Vaadin {@link com.vaadin.data.Property} value.
018 *
019 * <p>
020 * This annotation indicates that a read-only Vaadin {@link com.vaadin.data.Property} having the {@linkplain #value specified name}
021 * and type derived from the annotated field or zero-argument method is accessible by reading that field or method.
022 * </p>
023 *
024 * <p>
025 * Classes with {@link ProvidesProperty &#64;ProvidesProperty}-annotated fields and methods can be used to automatically
026 * generate a {@link PropertyExtractor} and a list of {@link PropertyDef}s by using a {@link PropertyReader}.
027 * </p>
028 *
029 * <p>
030 * The {@link com.vaadin.data.Container} classes in this package that are configured with a {@link PropertyExtractor}
031 * and a list of {@link PropertyDef}s all have a constructor variant taking a {@link ProvidesProperty
032 * &#64;ProvidesProperty}-annotated Java class, which allows the container properties to be auto-detected via introspection.
033 * </p>
034 *
035 * <p>
036 * Some details regarding {@link ProvidesProperty &#64;ProvidesProperty} annotations on methods:
037 *  <ul>
038 *  <li>Only non-void methods taking zero parameters are supported; {@link ProvidesProperty &#64;ProvidesProperty}
039 *      annotations on other methods are ignored</li>
040 *  <li>{@link ProvidesProperty &#64;ProvidesProperty} annotations on interface methods are supported</li>
041 *  <li>If a method and the superclass or superinterface method it overrides are both annotated with
042 *      {@link ProvidesProperty &#64;ProvidesProperty}, but the annotations declare different properties
043 *      (i.e., they have different {@linkplain #value names}), then two properties will be defined with the same value</li>
044 *  <li>After introspection, all property names must be unique, with the exception that a method may redundantly
045 *      declare the same property (i.e., having the same {@linkplain #value name}) as any method it overrides.</li>
046 *  </ul>
047 * </p>
048 *
049 * @see PropertyReader
050 */
051@Retention(RetentionPolicy.RUNTIME)
052@Target({ ElementType.METHOD, ElementType.FIELD })
053@Documented
054public @interface ProvidesProperty {
055
056    /**
057     * Get the name of the Vaadin property. If this is left unset (empty string), then the name of the
058     * annotated field, or the bean property name of the annotated "getter" method is used.
059     */
060    String value() default "";
061}
