008package org.dellroad.stuff.vaadin7;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotates a Java field or method that provides a read-only Vaadin {@link com.vaadin.data.Property} value.
018 *
019 * <p>
020 * This annotation indicates that a read-only Vaadin {@link com.vaadin.data.Property} having the {@linkplain #value specified name}
021 * and type derived from the annotated field or zero-argument method is accessible by reading that field or method.
022 * </p>
023 *
024 * <p>
025 * Classes with {@link ProvidesProperty &#64;ProvidesProperty}-annotated fields and methods can be used to automatically
026 * generate a {@link PropertyExtractor} and a list of {@link PropertyDef}s by using a {@link PropertyReader}.
027 * </p>
028 *
029 * <p>
030 * The {@link com.vaadin.data.Container} classes in this package that are configured with a {@link PropertyExtractor}
031 * and a list of {@link PropertyDef}s all have a constructor variant taking a {@link ProvidesProperty
032 * &#64;ProvidesProperty}-annotated Java class, which allows the container properties to be auto-detected via introspection.
033 * For example:
034 * <blockquote><pre>
035 * // Container backing object class
036 * public class User {
037 *
038 *     public static final String USERNAME_PROPERTY = "username";
039 *     public static final String REAL_NAME_PROPERTY = "realName";
040 *
041 *     private String username;
042 *     private String realName;
043 *
044 *     public String getUsername() {
045 *         return this.username;
046 *     }
047 *     public void setUsername(String username) {
048 *         this.username = username;
049 *     }
050 *
051 *     &#64;ProvidesProperty(REAL_NAME_PROPERTY)
052 *     public String getRealName() {
053 *         return this.realName;
054 *     }
055 *     public void setRealName(String realName) {
056 *         this.realName = realName;
057 *     }
058 *
059 *     &#64;ProvidesProperty(USERNAME_PROPERTY)
060 *     public Label usernameProperty() {
061 *         return new Label("&lt;code&gt;" + StringUtil.escapeHtml(this.username) + "&lt;/code&gt;", ContentMode.HTML);
062 *     }
063 * }
064 *
065 * // User container class
066 * public class UserContainer extends SimpleKeyedContainer&lt;String, User&gt; {
067 *
068 *     public UserContainer() {
069 *         super(User.class);
070 *     }
071 *
072 *     &#64;Override
073 *     public String getKeyFor(User user) {
074 *         return user.getUsername();
075 *     }
076 * }
077 *
078 * // Create container holding all users
079 * UserContainer container = new UserContainer();
080 * container.load(this.getAllUsers());
081 *
082 * // Build table showing users
083 * Table table = new Table();
084 * table.setColumnHeader(User.USERNAME_PROPERTY, "User");
085 * table.setColumnHeader(User.REAL_NAME_PROPERTY, "Name");
086 * table.setVisibleColumns(User.USERNAME_PROPERTY, User.REAL_NAME_PROPERTY);
087 *
088 * // Select user "jsmith" in the table
089 * table.setValue("jsmith");
090 * ...
091 * </pre></blockquote>
092 * </p>
093 *
094 * <p>
095 * Some details regarding {@link ProvidesProperty &#64;ProvidesProperty} annotations on methods:
096 *  <ul>
097 *  <li>Only non-void methods taking zero parameters are supported; {@link ProvidesProperty &#64;ProvidesProperty}
098 *      annotations on other methods are ignored</li>
099 *  <li>{@link ProvidesProperty &#64;ProvidesProperty} annotations on interface methods are supported</li>
100 *  <li>If a method and the superclass or superinterface method it overrides are both annotated with
101 *      {@link ProvidesProperty &#64;ProvidesProperty}, but the annotations declare different properties
102 *      (i.e., they have different {@linkplain #value names}), then two properties will be defined with the same value</li>
103 *  <li>After introspection, all property names must be unique, with the exception that a method may redundantly
104 *      declare the same property (i.e., having the same {@linkplain #value name}) as any method it overrides.</li>
105 *  </ul>
106 * </p>
107 *
108 * @see PropertyReader
109 */
110@Retention(RetentionPolicy.RUNTIME)
111@Target({ ElementType.METHOD, ElementType.FIELD })
112@Documented
113public @interface ProvidesProperty {
114
115    /**
116     * Get the name of the Vaadin property. If this is left unset (empty string), then the name of the
117     * annotated field, or the bean property name of the annotated "getter" method is used.
118     */
119    String value() default "";
120}
